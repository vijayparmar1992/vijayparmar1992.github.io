    backToTopBtn.style.opacity = '0.3';
    backToTopBtn.style.pointerEvents = 'none';
  }
});

// Initialize
document.addEventListener('DOMContentLoaded', () => {
  observeElements();
});

// Keyboard shortcuts
document.addEventListener('keydown', (e) => {
  if ((e.ctrlKey || e.metaKey) && e.key === 'd') {
    e.preventDefault();
    themeToggle.click();
  }
  if (e.key === 'Home' && e.ctrlKey) {
    e.preventDefault();
    scrollToTop();
  }
});
